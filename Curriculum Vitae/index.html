Script Global – Adventure Game Studio

Este script controla la lógica principal del juego: gestión de entradas del jugador, comportamiento de personajes, inventario y transiciones entre habitaciones. Es una parte esencial del sistema de interacción del proyecto. 


// main global script file

int lastFlipTime = 0; // Para llevar el control del tiempo
int flipInterval = 10000; // 10 segundos en milisegundos
int isDialogActive = 0;

// Definir una variable global para controlar la dirección
int enemigoDireccion = 1;  // 1 significa que está mirando a la derecha (eDirectionRight), 0 para izquierda (eDirectionLeft)

// called when the game starts, before the first room is loaded
function game_start()
{
  SetRestartPoint();
  mouse.Visible = true;
  mouse.Mode = eModeWalkto;
}

function repeatedly_execute()
{
  if (isDialogActive == 1) return;
  LocationType location = GetLocationType(mouse.x, mouse.y);
  if (location == eLocationObject || location == eLocationCharacter) {
    mouse.Mode = eModeUsermode1;
  } else {
    mouse.Mode = eModeWalkto;
  }
}

function on_key_press(eKeyCode keycode, int mod)
{
  if (IsGamePaused()) keycode = 0;
  else if (keycode == eKeyQ && (mod & eKeyModCtrl)) QuitGame(1);
  else if (keycode == eKeyF9) RestartGame();
  else if (keycode == eKeyF12) SaveScreenShot("screenshot.pcx");
}
